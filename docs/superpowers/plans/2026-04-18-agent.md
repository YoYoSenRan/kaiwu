# Agent 模块 Implementation Plan

**Goal:** 实现 kaiwu agent 模块 — 本地 3 列索引表 + openclaw gateway 真实数据源 + list 三分区 + detail 四 tab.

**Architecture:** kaiwu `agents` 表做引用索引, 所有字段数据实时从 openclaw RPC 拉取. 列表派生 mine/unsynced/missing. 详情一次聚合 RPC 拉全 (identity + files + skills + tools).

**Tech Stack:** electron main (ts-ioc style decorators) + drizzle-orm + better-sqlite3 + React 19 + zustand + sonner (toast) + i18next.

**Note:** 每个任务不做 git commit, 等全部完成用户自己提交.

```ts
export interface AgentRow {

  createdAt: number
  updatedAt: number
}

export type AgentListStatus = "mine" | "unsynced" | "missing"

export interface AgentListEntry {
  agentId: string

  // 若 status = mine/unsynced, 会有 openclaw 侧数据
  gateway?: GatewayAgentRow
  // 若 status = mine/missing, 会有 kaiwu 侧数据
  local?: AgentRow
}

export interface AgentListResult {
  gatewayReady: boolean
  gatewayEmpty: boolean
  defaultId?: string
  mine: AgentListEntry[]
  unsynced: AgentListEntry[]
  missing: AgentListEntry[]
}

export interface AgentCreateInput {
  name: string
  workspace: string
  model?: string
  emoji?: string
  avatar?: string
}

export interface AgentUpdateInput {
  agentId: string
  name?: string
  workspace?: string
  model?: string
  emoji?: string
  avatar?: string
}

export type AgentDeleteStrategy = { kind: "purge"; deleteFiles: boolean } | { kind: "unlink" }

export interface AgentDeleteInput {
  agentId: string
  strategy: AgentDeleteStrategy
}

export interface AgentImportInput {
  agentIds: string[]
}

export interface AgentDetail {
  agentId: string
  gateway?: GatewayAgentRow
  identity?: AgentIdentityGetResult
  files?: AgentFilesListResult
  skills?: SkillsStatusResult
  toolsCatalog?: ToolsCatalogResult
  toolsEffective?: ToolsEffectiveResult

  update: (p: AgentUpdateInput) => Promise<{ ok: true }>
```

```ts
import { eq, inArray } from "drizzle-orm"
import { db } from "../../database/client"
import { agents } from "../../database/schema"
import type { AgentRow } from "./types"

export function listAll(): AgentRow[] {
  return db.select().from(agents).all().map(toRow)
}

export function get(agentId: string): AgentRow | null {
  const row = db.select().from(agents).where(eq(agents.agentId, agentId)).get()
  return row ? toRow(row) : null
}

export function insert(agentId: string): AgentRow {
  const now = Date.now()
  db.insert(agents).values({ agentId, createdAt: now, updatedAt: now }).run()
  return { agentId, createdAt: now, updatedAt: now }
}

export function insertMany(agentIds: string[]): number {
  if (agentIds.length === 0) return 0
  const now = Date.now()
  const rows = agentIds.map((id) => ({ agentId: id, createdAt: now, updatedAt: now }))
  // INSERT OR IGNORE via drizzle onConflictDoNothing
  db.insert(agents).values(rows).onConflictDoNothing().run()
  return agentIds.length
}

export function touch(agentId: string): void {
  db.update(agents).set({ updatedAt: Date.now() }).where(eq(agents.agentId, agentId)).run()
}

export function remove(agentId: string): void {
  db.delete(agents).where(eq(agents.agentId, agentId)).run()
}

function toRow(raw: typeof agents.$inferSelect): AgentRow {
  return { agentId: raw.agentId, createdAt: raw.createdAt, updatedAt: raw.updatedAt }
}
```

## Task C3: kaiwu agent service (@Controller)

**Files:**

- Create: `electron/features/agent/service.ts`

Steps:

- [ ] service 设计:

```ts
import { Controller, Handle } from "../../app/ipc"
import { getGateway } from "../openclaw/runtime"
import { agents as agentsDomain } from "../openclaw/domains/agents"
import * as repo from "./repository"
import type { AgentCreateInput, AgentDeleteInput, AgentDetail, AgentImportInput, AgentListEntry, AgentListResult, AgentUpdateInput } from "./types"

@Controller("agent")
export class AgentService {
  @Handle("list")
  async list(): Promise<AgentListResult> {
    const gw = getGateway()
    const local = repo.listAll()
    const localIds = new Set(local.map((r) => r.agentId))

    if (!gw || !gw.isConnected()) {
      return {
        gatewayReady: false,
        gatewayEmpty: false,
        mine: [],
        unsynced: [],
        missing: local.map((r) => ({ agentId: r.agentId, status: "missing" as const, local: r })),
      }
    }

    const res = await agentsDomain.methods.list(gw)
    const gatewayIds = new Set(res.agents.map((a) => a.id))
    const gatewayEmpty = res.agents.length === 0

    const mine: AgentListEntry[] = []
    const unsynced: AgentListEntry[] = []
    const missing: AgentListEntry[] = []

    for (const g of res.agents) {
      if (localIds.has(g.id)) {
        mine.push({ agentId: g.id, status: "mine", gateway: g, local: local.find((l) => l.agentId === g.id) })
      } else {
        unsynced.push({ agentId: g.id, status: "unsynced", gateway: g })
      }
    }
    if (!gatewayEmpty) {
      for (const l of local) {
        if (!gatewayIds.has(l.agentId)) {
          missing.push({ agentId: l.agentId, status: "missing", local: l })
        }
      }
    }

    return { gatewayReady: true, gatewayEmpty, defaultId: res.defaultId, mine, unsynced, missing }
  }

  @Handle("detail")
  async detail(agentId: string): Promise<AgentDetail> {
    const gw = getGateway()
    if (!gw || !gw.isConnected()) return { agentId }

    const [listRes, identity, files, skills, toolsCatalog, toolsEffective] = await Promise.allSettled([
      agentsDomain.methods.list(gw),
      agentsDomain.methods.identity(gw, { agentId }),
      agentsDomain.methods.filesList(gw, { agentId }),
      agentsDomain.methods.skillsStatus(gw, { agentId }),
      agentsDomain.methods.toolsCatalog(gw, { agentId, includePlugins: true }),
      agentsDomain.methods.toolsEffective(gw, { agentId }),
    ])

    const gateway = listRes.status === "fulfilled" ? listRes.value.agents.find((a) => a.id === agentId) : undefined
    return {
      agentId,
      gateway,
      identity: identity.status === "fulfilled" ? identity.value : undefined,
      files: files.status === "fulfilled" ? files.value : undefined,
      skills: skills.status === "fulfilled" ? skills.value : undefined,
      toolsCatalog: toolsCatalog.status === "fulfilled" ? toolsCatalog.value : undefined,
      toolsEffective: toolsEffective.status === "fulfilled" ? toolsEffective.value : undefined,
    }
  }

  @Handle("create")
  async create(input: AgentCreateInput): Promise<{ agentId: string }> {
    const gw = getGateway()
    if (!gw) throw new Error("gateway not connected")
    const res = await agentsDomain.methods.create(gw, input)
    repo.insert(res.agentId)
    return { agentId: res.agentId }
  }

  @Handle("update")
  async update(input: AgentUpdateInput): Promise<{ ok: true }> {
    const gw = getGateway()
    if (!gw) throw new Error("gateway not connected")
    await agentsDomain.methods.update(gw, input)
    repo.touch(input.agentId)
    return { ok: true }
  }

  @Handle("delete")
  async delete(input: AgentDeleteInput): Promise<{ ok: true; removedBindings?: number }> {
    const gw = getGateway()
    if (input.strategy.kind === "unlink") {
      repo.remove(input.agentId)
      return { ok: true }
    }
    if (!gw) throw new Error("gateway not connected")
    const res = await agentsDomain.methods.delete(gw, {
      agentId: input.agentId,
      deleteFiles: input.strategy.deleteFiles,
    })
    repo.remove(input.agentId)
    return { ok: true, removedBindings: res.removedBindings }
  }

  @Handle("importUnsynced")
  async importUnsynced(input: AgentImportInput): Promise<{ imported: number }> {
    const n = repo.insertMany(input.agentIds)
    return { imported: n }
  }

  @Handle("filesGet")
  async filesGet(p: { agentId: string; name: string }) {
    const gw = getGateway()
    if (!gw) throw new Error("gateway not connected")
    const res = await agentsDomain.methods.filesGet(gw, p)
    return { file: res.file }
  }

  @Handle("filesSet")
  async filesSet(p: { agentId: string; name: string; content: string }): Promise<{ ok: true }> {
    const gw = getGateway()
    if (!gw) throw new Error("gateway not connected")
    await agentsDomain.methods.filesSet(gw, p)
    return { ok: true }
  }
}
```

```ts
export function useAgentList() {
  const [data, setData] = useState<AgentListResult | null>(null)

  const refresh = useCallback(async () => {
```
